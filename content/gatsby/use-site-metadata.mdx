<Text>No need to use the StaticQuery component anymore. Simplify things with this hook. </Text>

```js
import { useStaticQuery, graphql } from "gatsby";

export const useSiteMetadata = () => {
  const { site } = useStaticQuery(
    graphql`
      query SiteMetaData {
        site {
          siteMetadata {
            title
            description
            ...allTheRest
          }
        }
      }
    `
  );
  return site.siteMetadata;
};
```
